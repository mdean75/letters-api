// Comprehensive test with nested case

function main aa: integer, b: real returns real;
    c: integer is 8;
    d: real is 7.E2;
begin
    case aa is
         when 1 => aa * 2 / d ** 2;
         when 2 => aa + 5.E+2 - b;
         when 3 =>
             case d is
                 when 1 => aa rem 2;
                 others => 9.1E-1;
             endcase;
         when 4 => aa / 2 - c;
         others => aa + 4.7 * b;
    endcase;
end;
